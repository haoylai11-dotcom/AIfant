🌐 浏览器页面字段提取

使用方式

  1. 在已登录的浏览器中打开抖音或快手公开视频页面
  2. 打开浏览器开发者工具 (F12) → Console
  3. 粘贴下方JavaScript脚本并运行
  4. 将输出的JSON结果复制到下方文本框
  5. 点击"导入"
📋 浏览器提取脚本（点击展开） 
// === 复制以下全部内容到浏览器 Console ===
(function() {
  const result = {
    url: window.location.href,
    title: document.title,
    platform: window.location.hostname.includes('douyin.com') ? 'douyin' : 'kuaishou',
    // 以下字段请根据页面实际显示内容手工补充：
    description: '',
    hashtags: [],
    publish_time: '',
    duration: 0,
    author_name: '',
    author_id: '',
    follower_count: null,
    verified: false,
    like_count: null,
    comment_count: null,
    share_count: null,
    favorite_count: null,
    view_count: null,
    comments: [],
    search_keyword: '',
    search_rank: null,
    sort_mode: 'comprehensive'
  };
  console.log(JSON.stringify(result, null, 2));
  return result;
})();
